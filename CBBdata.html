big10 <- teamboxscores |> filter(team_short_display_name %in% c("Nebraska", "Michigan", "Iowa", "Michigan St", "Penn State", "Wisconsin", "Minnesota", "Illinois", "USC", "UCLA", "Washington", "Oregon", "Purdue", "Northwestern","Maryland", "Rutgers", "Ohio State", "Indiana"))
bigcbb <- big10 |>
  select(team_short_display_name, assists, team_turnovers) |>
  mutate_at(vars(-team_short_display_name), as.numeric) |>
  group_by(team_short_display_name) |>
  summarize(
    AvgAssists = mean(assists),
    AvgTurnovers = mean(team_turnovers),
  ) |>
  rename(Team = team_short_display_name) |>
  select(Team, AvgAssists, AvgTurnovers) |>
  arrange(desc(AvgAssists))

 big10lead <- big10 |> 
  filter(season_type == 2) |> 
  filter(is.na(lead_changes) == FALSE) |> 
  group_by(team_short_display_name)|> 
  arrange(game_date) |>
  mutate(cum_lead_change = cumsum(as.numeric(lead_changes)))
mn <- big10lead |> filter(team_short_display_name == "Minnesota")
mi <- big10lead |> filter(team_short_display_name == "Michigan")
ne <- big10lead |> filter(team_short_display_name == "Nebraska")
ggplot() + geom_step(data=big10lead, aes(x=game_date, y=cum_lead_change, group=team_short_display_name), color="lightgrey") +
  geom_step(data=mn, aes(x=game_date, y=cum_lead_change, group=team_short_display_name), color="#7A0019") +
